31 bool XBot::AdvrJointState::init(const std::string& path_to_config_file, GenericJointStateMessage::Type type)
32 {
33 
34  std::cout << "Initializing AdvrJointState message interface!" << std::endl;
35 
36  YAML::Node root_cfg = YAML::LoadFile(path_to_config_file);
37 
38  // TBD check if they exist
39  const YAML::Node &jointstate_root = root_cfg["AdvrJointStateMessage"];
40  if(!jointstate_root){
41  std::cerr << "ERROR in " << __func__ << "! Provided config file does not contain mandatory node \"JointState\" " << std::endl;
42  return false;
43  }
44 
45  _topic_name = jointstate_root["joint_state_topic_name"].as<std::string>();
46 
47  ros::NodeHandle nh;
48  _sub = nh.subscribe(_topic_name, 1, &AdvrJointState::callback, this);
49 
50 
51  // Max number of attempts to connect to /joint_states topic
52  int max_connect_attempts = 100;
53  double sleep_duration = 0.01;
54  int attempt = 0;
55 
56 
57  if( type == GenericJointStateMessage::Type::Rx ){
58  // Attempt to get a joint state message
59  while (!_msg_received && attempt < max_connect_attempts) {
60  ros::spinOnce();
61  ros::Duration(sleep_duration).sleep();
62  attempt++;
63  }
64 
65  if (!_msg_received) {
66  std::cerr << "ERROR No message received on topic " << _topic_name << "! Read() won't work!!" << std::endl;
67  return false;
68  }
69 
70  }
71 
72  if( type == GenericJointStateMessage::Type::Tx ){
73 
74  // Choosing a random joint order for publishing
75  auto robot = XBot::RobotInterface::getRobot(path_to_config_file);
76 
77  _msg.effort.clear();
78  _msg.link_position.clear();
79  _msg.link_velocity.clear();
80  _msg.motor_position.clear();
81  _msg.motor_velocity.clear();
82  _msg.aux.clear();
83  _msg.stiffness.clear();
84  _msg.damping.clear();
85  _msg.name.clear();
86  _msg.temperature.clear();
87 
88 
89  for( const std::string& jname : robot->getEnabledJointNames() ){
90  _msg.effort.push_back(0);
91  _msg.link_position.push_back(0);
92  _msg.link_velocity.push_back(0);
93  _msg.motor_position.push_back(0);
94  _msg.motor_velocity.push_back(0);
95  _msg.aux.push_back(0);
96  _msg.stiffness.push_back(0);
97  _msg.damping.push_back(0);
98  _msg.position_reference.push_back(0);
99  _msg.velocity_reference.push_back(0);
100  _msg.effort_reference.push_back(0);
101  _msg.temperature.push_back(0);
102  _msg.name.push_back(jname);
103  }
104 
105 
106  }
107 
108 
109  // Populate _idx_map
110  {
111  int idx = 0;
112  for( const std::string& joint_name : _msg.name){
113  _id_map[joint_name] = idx;
114  idx++;
115  }
116 
117  }
118 
119  // Get a publisher
120  _pub = nh.advertise<XCM::JointStateAdvr>(_topic_name, 1);
121 
122  return true;
